local function C_11()
local script = G2L["11"];
	local UIS = game:GetService('UserInputService')
	local frame = script.Parent
	
	local dragToggle = nil
	local dragSpeed = 0.01
	local dragStart = nil
	local startPos = nil
	
	local function updateInput(input)
		local delta = input.Position - dragStart
		local position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X,
			startPos.Y.Scale, startPos.Y.Offset + delta.Y) 
		game:GetService('TweenService') :Create(frame, TweenInfo.new(dragSpeed), {Position = position}) :Play()
	end
	
	frame.InputBegan:Connect(function(input)
		if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
			dragToggle = true
			dragStart = input.Position
			startPos = frame.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragToggle = false
				end
			end)
		end
	end)
	
	UIS.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			if dragToggle then
				updateInput(input)
			end
		end
	end)
	
end;
task.spawn(C_11);

local function C_19()
local script = G2L["19"];
	local button = script.Parent
	local isActive = false
	local textLabel = script.Parent.EnabledDisabled
	
	local originalPosition = textLabel.Position
	local newPosition = UDim2.new(originalPosition.X.Scale + 0.55, originalPosition.X.Offset, originalPosition.Y.Scale, originalPosition.Y.Offset)
	
	local function updateAttributes()
		local player = game:GetService("Players").LocalPlayer
		local backpack = player.Backpack
	
		for _, tool in pairs(backpack:GetChildren()) do
			local ammoAttribute = tool:GetAttribute("Ammo")
			if ammoAttribute then
				tool:SetAttribute("Ammo", isActive and 9999 or 5)
			end
		end
	end
	
	local function updateAttributes1()
		local player = game:GetService("Players").LocalPlayer
		local char = player.Character
	
		for _, tool in pairs(char:GetChildren()) do
			local ammoAttribute = tool:GetAttribute("Ammo")
			if ammoAttribute then
				tool:SetAttribute("Ammo", isActive and 9999 or 5)
			end
		end
	end
	
	local function startLoop()
		while isActive do
			wait(0.1)
			updateAttributes()
			updateAttributes1()
			textLabel:TweenPosition(newPosition, Enum.EasingDirection.Out, Enum.EasingStyle.Quart, 0.5, true)
			textLabel.BackgroundColor3 = Color3.fromRGB(43, 59, 36)
		end
		textLabel:TweenPosition(originalPosition, Enum.EasingDirection.Out, Enum.EasingStyle.Quart, 0.5, true)

local function C_32()
local script = G2L["32"];
	local button = script.Parent
	local GUI = script.Parent.Parent.Parent.Parent
	
	local frame = script.Parent.Parent.Parent
	local transparency = 0
	local transparencyStep = 0.1
	
	local soundId = "rbxassetid://9118052301"
	
	local soundInstance = Instance.new("Sound")
	soundInstance.SoundId = soundId
	soundInstance.Parent = button
	soundInstance.PlayOnRemove = true
	
	local isPlaying = false
	
	local function fadeOut()
		while transparency < 0.8 do
			transparency = transparency + transparencyStep
			frame.BackgroundTransparency = transparency
			wait(0.1)
		end
		GUI:Destroy()
	end
	
	button.MouseButton1Click:Connect(function()
		if not isPlaying then
			isPlaying = true
			soundInstance:Play()
			fadeOut()
		end
	end)
end;
